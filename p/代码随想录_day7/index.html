class Solution {
public:
    vector<vector<int>> one_fourSum(vector<int> &nums, int target) {

        vector<vector<int>> res_vec; // vectoråˆå§‹åŒ–çš„æ•°ç»„ä¸ºç©º

        for (int i = 0; i < nums.size() - 3; i++) {

            // target =0çš„æƒ…å†µä¸‹, num[i]=target=0,åç»­çš„å€¼ä¹Ÿå¯ä»¥æ‰¾åˆ°å››å…ƒç»„,

            // æ€»ç»“
            // target=0 , nums[i] > target
            // target >0çš„æƒ…å†µ, nums[i] >=target

            if (target == 0 && nums[i] > target ||
                target > 0 && nums[i] >= target) {
                break;
            }
            // 1.2 iå»é‡
            // ié‡å¤çš„æƒ…å†µä¸‹, ç»“æŸæœ¬æ¬¡å¾ªç¯
            if (i > 0 && nums[i] == nums[i - 1]) {
                continue;
            }
            for (int j = i + 1; j < nums.size() - 2; j++) {
                // 2.1 j å‰ªæ
                // åœ¨æœ‰içš„åŸºç¡€ä¸Š, å¦‚æœtarge >= 0, nums[i] + nums[j] >= target
                // >=0 ,ä¹Ÿå°±æ„å‘³ç€ä¸å¯èƒ½æ˜¯ç›®æ ‡çš„å››å…ƒç»„
                if (target == 0 && nums[i] + nums[j] > target ||
                    target > 0 && nums[i] + nums[j] >= target) {
                    break;
                }

                // 2.2 jå»é‡
                // j > 1 && nums[j] == nums[j - 1])

                // å®é™…ä¸Šè¿™é‡Œè¦é¿å…çš„æ˜¯iå’ŒjåŒæ—¶å’Œä¸Šä¸€è½®çš„ç›¸åŒ
                // ä¹Ÿå°±æ˜¯ç»“æŸå¾ªç¯çš„æ¡ä»¶æ˜¯j>1, iå’Œä¸Šä¸€è½®çš„ç›¸åŒä¸”jå’Œä¸Šä¸€è½®ç›¸åŒ

                if (j > i + 1 && nums[j] == nums[j - 1]) {

                int left = j + 1;

                while (left < right) {
                    long long temp =
                        (long long)nums[i] + nums[j] + nums[left] + nums[right];

                            left++;
                        } while (nums[left] == nums[left - 1] && left < right);

                        do {

                                 left < right);

    vector<vector<int>> fourSum(vector<int> &nums, int target) {
        vector<vector<int>> res_vec;
        if (nums.size() < 4)
            return res_vec;
        sort(nums.begin(), nums.end());
        size_t length = nums.size();
        for (int i = 0; i < length - 3; i++) {
            // 1.å‰ªæ
            // æœ€å°çš„ç»„åˆéƒ½å¤§äºtarget ,æ— è§£, break
            if ((long long)nums[i] + nums[i + 1] + nums[i + 2] + nums[i + 3] >
                target) {

            // åˆå§‹å€¼åŠ æœ€å¤§çš„ä¸‰ä¸ªç»„åˆéƒ½å°äºtarget, è¯´æ˜å½“å‰å€¼è‚¯å®šä¸è¡Œ
            if ((long long)nums[i] + nums[length - 3] + nums[length - 2] +
                    nums[length - 1] <
                target) {
                continue;
            }
            if (i > 0 && nums[i] == nums[i - 1]) {
                continue;
            }

            for (int j = i + 1; j < length - 2; j++) {
                if ((long long)nums[i] + nums[j] + nums[j + 1] + nums[j + 2] >
                    target) {
                    break;
                }

                if ((long long)nums[i] + nums[j] + nums[length - 2] +
                        nums[length - 1] <
                    target) {
                    continue;
                }

                if (j > i + 1 && nums[j] == nums[j - 1]) {
                    continue;
                }

                int left = j + 1;
                int right = nums.size() - 1;
                while (left < right) {
                    long long sum =
                        (long long)nums[i] + nums[j] + nums[left] + nums[right];
                    if (sum > target) {
                        right--;
                    } else if (sum < target) {
                        left++;
                    } else {
                        res_vec.push_back(
                            {nums[i], nums[j], nums[left], nums[right]});
                        do {
                            left++;
                        } while (left < right && nums[left] == nums[left - 1]);
                        do {
                            right--;
                        } while (left < right &&
                                 nums[right] == nums[right + 1]);
                    }
                }
            }
        }
        return res_vec;
    }
};
