        // ÈÅçÂéÜmap, ÂÖÉÁ¥†Âä†ÂÖ•Â∞èÈ°∂Â†Ü, Â§ß‰∫ék‰∏™ÂºπÂá∫
        // Â∞èÈ°∂Â†ÜÊéíÂ∫èÁöÑÂ§çÊùÇÂ∫¶o(logk)

        for (auto i : map_res) {

        vector<int> vec_res(k);
        for (int i = k - 1; i >= 0; i--) {
            vec_res[i] = queue_res.top().first;
            queue_res.pop();
        }
        return vec_res;
